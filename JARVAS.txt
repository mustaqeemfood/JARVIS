            width: 80%;

<body>
    <h1>JARVIS AI Assistant</h1>
    <div id="jarvis-output"></div>
    <button id="start-button">Start</button>
    <script>
        const startButton = document.getElementById('start-button');
        const jarvisOutput = document.getElementById('jarvis-output');
        const synth = window.speechSynthesis;

        startButton.addEventListener('click', () => {
            startButton.disabled = true;
            startButton.textContent = 'Listening...';
            startSpeechRecognition();
        });

        function startSpeechRecognition() {
            const recognition = new webkitSpeechRecognition() || new SpeechRecognition();
            recognition.maxResults = 10;
            recognition.onresult = event => {
                const transcript = event.results[0][0].transcript;
                jarvisOutput.textContent = `You said: ${transcript}`;
                processCommand(transcript);
            };
            recognition.onerror = event => {
                console.error('Error occurred:', event);
            };
            recognition.onend = () => {
                startButton.disabled = false;
                startButton.textContent = 'Start';
            };
            recognition.start();
        }

        function processCommand(command) {
            // Simple command processing
            if (command.toLowerCase().includes('hello')) {
                jarvisOutput.textContent = 'Hello, Sir!';
                speakText('Hello, Sir!');
            } else if (command.toLowerCase().includes('time')) {
                const currentTime = new Date().toLocaleTimeString();
                jarvisOutput.textContent = `The current time is ${currentTime}`;
                speakText(`The current time is ${currentTime}`);
            } else if (command.toLowerCase().includes('date')) {
                const currentDate = new Date().toLocaleDateString();
                jarvisOutput.textContent = `Today's date is ${currentDate}`;
                speakText(`Today's date is ${currentDate}`);
            } else if (command.toLowerCase().includes('what is your name')) {
                jarvisOutput.textContent = 'My name is JARVIS';
                speakText('My name is JARVIS');
            } else if (command.toLowerCase().includes('exit')) {
                jarvisOutput.textContent = 'Goodbye, Sir!';
                speakText('Goodbye, Sir!');
            } else {
                jarvisOutput.textContent = 'Unknown command. Please try again!';
                speakText('Unknown command. Please try again!');
            }
        }

        function speakText(text) {
            if (synth.speaking) {
                synth.cancel();
            }
            const utterance = new SpeechSynthesisUtterance(text);
            synth.speak(utterance);
        }
    </script>
</body>
